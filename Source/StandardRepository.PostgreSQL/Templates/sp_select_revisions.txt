CREATE OR REPLACE FUNCTION ##FULL_TABLE_NAME##_select_revisions
(
    prm_##TABLE_NAME##_id bigint	
)
RETURNS TABLE
(
##ALL_FIELDS_INCLUDING_REVISION_FIELDS_WITH_TYPE##
)
AS $$
BEGIN

    RETURN QUERY
    SELECT  ##ALL_FIELDS_INCLUDING_REVISION_FIELDS_WITH_PREFIX##
    FROM ##FULL_TABLE_NAME##_revision t
    WHERE t.##TABLE_NAME##_id = prm_##TABLE_NAME##_id AND t.is_deleted = false;

END
$$ LANGUAGE 'plpgsql';
